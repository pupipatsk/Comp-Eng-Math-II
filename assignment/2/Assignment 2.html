t = np.linspace(-4, 4, 1000)

def x(t):
    return np.where((t >= -1) & (t <= 1), 1, 0)

def h(t):
    return np.where((t >= 0) & (t <= 1), 1, 0) - np.where((t >= -1) & (t <= 0), 1, 0)

y = np.convolve(x(t), h(t), mode='same') * (t[1] - t[0]) # Normalize by time step

def plot(t, x, h, y):
    plt.figure(figsize=(16, 9))
    plt.plot(t, x(t), label='x(t)', alpha=0.5)
    plt.plot(t, h(t), label='h(t)', color='orange', alpha=0.5)
    plt.plot(t, y, label='y(t) = x(t) * h(t)', color='green')
    plt.title('Convolution')
    plt.xlabel('t')
    plt.ylabel('Amplitude')
    plt.grid(True)
    plt.legend()
    plt.show()

plot(t, x, h, y)

# 3.

t = np.linspace(-4, 4, 1000)

def x(t):
    rect = np.where((t >= -1) & (t <= 0), 2, 0)

    idx = np.argsort(np.abs(t - 2))[:2]
    delta = np.zeros_like(t)
    delta[idx] = 1 # Unit impulse

    return rect + delta

def h(t):
    return np.where((t >= -1) & (t <= 1), 1, 0)

y = np.convolve(x(t), h(t), mode='same') * (t[1] - t[0]) # Normalize by time step

plot(t, x, h, y)
